1. 解答例 
/* bmi.c */

#include <stdio.h>

int main()
{
	double bl, bw, bmi;

	printf("身長(cm) :");
	scanf("%lf", &bl);
	printf("体重(kg) :");
	scanf("%lf", &bw);

	bmi = bw / ((bl / 100.0) * (bl / 100.0));
	printf("あなたのBMIは%fです\n", bmi);
	if (bmi >= 25.0)
		printf("肥満に注意!\n");

	return 0;
}

scanf関数で、double型の数値を受け取るには変換仕様は%lfとなります。float型の場合は%fです。一方printf関数でdouble型やfloat型の数値を表示する変換仕様は%fです。scanf関数とprintf関数では変換仕様が若干異なる点に注意してください。

		printf("点数 :");
		scanf("%d", &point);
		if (point == -1)
			break;
		sum += point;
		sum2 += point * point;
		n++;
	}

	mean = (double)sum / (double)n;
	sd = sqrt((sum2 - (mean) * (mean) * n)/(n - 1));

	printf("人数 = %d, 平均点 = %.1f点, 標準偏差 = %.1f\n",
		n, mean, sd);

	return 0;
}
