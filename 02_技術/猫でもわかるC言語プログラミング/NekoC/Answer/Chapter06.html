1. 解答例
/* memseki.c */

#include <stdio.h>

double menseki(double, double);

int main()
{
	double teihen, takasa;

	/* scanfとprintfでは%fの意味が違うことに注意*/
	printf("三角形の底辺は---");
	scanf("%lf", &teihen);
	printf("三角形の高さは---");
	scanf("%lf", &takasa);

	/*小数点以下2桁まで表示*/
	printf("三角形の面積は%.2fです\n"
		, menseki(teihen, takasa));

	return 0;
}

double menseki(double a, double h)
{
	return a * h * 0.5;
}

2. 解答例
/* tousa01.c */

#include <stdio.h>

int tosawa(int, int, int);

int main()
{
	int a, d, n, wa;

	printf("初項は---");
	scanf("%d", &a);
	printf("公差は---");
	scanf("%d", &d);
	printf("第何項までの和を求めますか---");
	scanf("%d", &n);

	wa = tosawa(a, d, n);
	printf("初項%d, 公差%dの第%d項までの等差数列の和は%dです\n",
		a, d, n, wa);

	return 0;
}

int tosawa(int a, int d, int n)
{
	int i, wa = 0;

	for (i = 0; i < n; i++)
		wa += a + i * d;

	return wa;
}

3. 解答例
/* shohizei.c */

#include <stdio.h>

double shohizei(int, double);

int main()
{
	int teika;
	double zei, zeikomi;

	printf("定価--");
	scanf("%d", &teika);
	printf("税率(%%)--");
	scanf("%lf", &zei);

	zeikomi = shohizei(teika, zei);

	printf("税込み価格は%.0f円です\n", zeikomi);

	return 0;
}

double shohizei(int a, double d)
{
	return a * (1 + d / 100.0);
}

4. 解答例
/* sekijun.c */

#include <stdio.h>

int junretsu(int n, int r);

int main()
{
	int n, r;

	printf("人数--");
	scanf("%d", &n);
	printf("席数--");
	scanf("%d", &r);

	printf("%d通り\n", junretsu(n, r));

	return 0;
}

int junretsu(n, m)
{
	int p = 1, i;

	for (i = m; i >= n; i--)
		p *= i;

	return p;
}
